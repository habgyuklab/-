        function toggleProcess(id, field, source) {
            let info = null;
            let dataItem = null;

            if (source === 'special') {
                dataItem = specialExceptions.find(s => s.id == id); // == ë¹„êµ (ë¬¸ì/ìˆ«ì í˜¸í™˜)
                if (dataItem) { info = dataItem.processState; }
            } else {
                dataItem = seatsData[id];
                if (dataItem) { info = dataItem.penaltyInfo; }
            }

            if (info) {
                info[field] = !info[field];
                saveData();
                
                if (info.recorded && info.noticed) {
                    finalizeProcess(dataItem, source);
                } else {
                    updateSidebarList(); 
                }
            }
        }

        function finalizeProcess(item, source) {
            const type = (source === 'special') ? item.processState.type : item.penaltyInfo.type;
            let seatId = (source === 'special') ? item.seatId : item.id;
            const name = item.name;

            if (!seatId || (seatId && seatsData[seatId] && seatsData[seatId].name !== name)) {
                 const foundSeat = getSeatDataByName(name);
                 if (foundSeat) seatId = foundSeat.id;
            }

            if (seatId && seatsData[seatId]) {
                if (type === 'COMPLETE') {
                    // ì™„ë£Œ -> ì´ˆë¡ìƒ‰(ì…ì‹¤)
                    seatsData[seatId].state = 1; 
                    seatsData[seatId].penaltyInfo = null;
                    seatsData[seatId].absentInfo = null;
                } else {
                    // ì¡°í‡´/ê²°ì„ -> íšŒìƒ‰(í‡´ì‹¤)
                    seatsData[seatId].state = 4; 
                    let reasonText = "";
                    if(source === 'special') reasonText = item.reason;
                    else if(item.penaltyInfo) reasonText = item.penaltyInfo.reason;

                    seatsData[seatId].absentInfo = {
                        type: type === 'EARLY' ? 'ì¡°í‡´' : 'ê²°ì„',
                        reason: reasonText,
                        time: (source === 'special') ? item.time : '23:59' 
                    };
                    seatsData[seatId].penaltyInfo = null; 
                }
            }

            if (source === 'special') {
                if (type === 'COMPLETE') {
                    specialExceptions = specialExceptions.filter(sp => sp.id !== item.id);
                }
            } 

            saveData();
            refreshVisuals();
            updateSidebarList();
        }
